public void setPositions(char[][] board){
	
	boardRowPos = 1;
	boardColPos = 5;
	
	alienRowNum = 0;
	alienColNum = 0;
	while ( alienRowNum < alienRowSize ){
			
		while ( alienColNum < alienColumnSize ){
			aliens[alienRowNum][alienColNum].alienYcoord = boardRowPos;
			aliens[alienRowNum][alienColNum].alienXcoord = boardColPos;
			board[boardRowPos][boardColPos] = "U";
			
			boardColPos++;
			alienColNum++;
		}
		
		boardRowPos++;
		boardColPos = 5;
		alienColNum = 0;
		
		alienRowNum++;
	}
}
